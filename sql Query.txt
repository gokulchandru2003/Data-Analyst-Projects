By Revenue
SELECT 
    product_name,
    SUM(Revenue) AS Total_Revenue
FROM sales_data
GROUP BY Product_Name
ORDER BY Revenue DESC
LIMIT 10;

By Quantity Sold
SELECT 
    Item Type,
    SUM(Unit_sold) AS Total_Unit
FROM sales_data
GROUP BY Product_Name
ORDER BY Total Unit DESC
LIMIT 10;

Monthly Revenue Trend
SELECT 
    DATE_FORMAT(STR_TO_DATE(order_date, '%m-%d-%y'),'%y-%m' AS month,
    SUM(Revenue) AS monthly_revenue
FROM sales_data
GROUP BY month
ORDER BY month


Region-wise Profit
SELECT 
    region,
    SUM(Profit) AS Total_Profit
FROM sales_data
GROUP BY Region
ORDER BY Total_Profit DESC;


Profit Margin by Region
SELECT 
    Region,
    SUM(profit) AS Total_Profit,
    SUM(Total Revenue) AS Total_Sales,
    (SUM(Profit) / SUM(sales_amount)) * 100 AS Profit_Margin_PCT
FROM sales_data
GROUP BY Region
ORDER BY Profit_Margin_PCT DESC;
